Simple Integration

Add Jeju Auth to your app in minutes

// Redirect to Jeju Auth
const authUrl = new URL('https://auth.jejunetwork.org/oauth/authorize')
authUrl.searchParams.set('client_id', YOUR_CLIENT_ID)
authUrl.searchParams.set('redirect_uri', YOUR_CALLBACK_URL)
authUrl.searchParams.set('response_type', 'code')

window.location.href = authUrl.toString()

// Handle callback
const code = new URLSearchParams(window.location.search).get('code')
const tokens = await fetch('https://auth.jejunetwork.org/oauth/token', {
  method: 'POST',
  body: JSON.stringify({ grant_type: 'authorization_code', code })
}).then(r => r.json())
